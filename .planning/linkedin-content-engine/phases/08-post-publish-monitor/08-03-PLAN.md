**4. Add the Incoming Comments card** AFTER the "Recent Comment Activity" card, still within the `<div className="mt-6">` section of the Engagement tab:

```tsx
{/* Incoming Comments on Your Posts */}
<Card className="mt-6">
  <CardHeader>
    <CardTitle className="text-heading-md flex items-center gap-2">
      <MessageCircle className="h-5 w-5" />
      Incoming Comments on Your Posts
    </CardTitle>
    {incomingComments.length > 0 && (
      <p className="text-sm text-muted-foreground">
        Recent comments from your audience -- classified with reply suggestions
      </p>
    )}
  </CardHeader>
  <CardContent>
    {incomingComments.length === 0 ? (
      <div className="text-center py-8 text-muted-foreground">
        <MessageCircle className="h-8 w-8 mx-auto mb-2 opacity-40" />
        <p className="text-sm">
          Incoming comments will appear here after the Post-Publish Monitor (WF7) starts running.
        </p>
      </div>
    ) : (
      <div className="space-y-3">
        {incomingComments.map((comment) => {
          const typeStyle = COMMENT_TYPE_STYLES[comment.comment_type || 'addition'] || COMMENT_TYPE_STYLES.addition;

          return (
            <div
              key={comment.id}
              className="p-3 rounded-lg bg-background-card-light border border-border"
            >
              {/* Comment header: commenter + type badge + time */}
              <div className="flex items-center justify-between mb-1.5">
                <div className="flex items-center gap-2">
                  <span className="text-sm font-medium text-foreground">
                    {comment.commenter_name || 'Unknown'}
                  </span>
                  <span className={`text-[10px] px-1.5 py-0.5 rounded font-medium ${typeStyle.bg} ${typeStyle.text}`}>
                    {typeStyle.label}
                  </span>
                  {comment.thread_state === 'follow_up_sent' && (
                    <span className="text-[10px] px-1.5 py-0.5 rounded bg-amber-100 dark:bg-amber-900/30 text-amber-700 dark:text-amber-400 font-medium">
                      Awaiting Reply
                    </span>
                  )}
                </div>
                <span className="text-xs text-muted-foreground">
                  {comment.detected_at
                    ? new Date(comment.detected_at).toLocaleDateString('en-US', { month: 'short', day: 'numeric', hour: 'numeric', minute: '2-digit' })
                    : ''}
                </span>
              </div>

              {/* Commenter headline */}
              {comment.commenter_headline && (
                <p className="text-xs text-muted-foreground mb-1">
                  {comment.commenter_headline}
                </p>
              )}

              {/* Comment text */}
              <p className="text-sm text-foreground mb-2">
                {comment.comment_text}
              </p>

              {/* Reply suggestion */}
              {comment.reply_suggestion && (
                <div className="p-2.5 rounded-lg bg-muted/50 border border-border/50">
                  <div className="flex items-center justify-between mb-1">
                    <span className="text-[10px] uppercase font-medium text-muted-foreground tracking-wide">
                      Suggested Reply
                    </span>
                    <button
                      onClick={() => handleCopyComment(comment.reply_suggestion!, `incoming-${comment.id}`)}
                      className="shrink-0 p-1 rounded-md hover:bg-muted transition-colors"
                      title="Copy reply to clipboard"
                    >
                      {copiedSuggestionId === `incoming-${comment.id}` ? (
                        <CheckCircle2 className="h-3.5 w-3.5 text-green-500" />
                      ) : (
                        <Copy className="h-3.5 w-3.5 text-muted-foreground" />
                      )}
                    </button>
                  </div>
                  <p className="text-sm text-foreground">
                    {comment.reply_suggestion}
                  </p>
                </div>
              )}

              {/* Comment engagement stats */}
              <div className="flex items-center gap-4 mt-2 text-xs text-muted-foreground">
                {comment.comment_likes > 0 && (
                  <span className="inline-flex items-center gap-1">
                    <ThumbsUp className="h-3 w-3" />
                    {comment.comment_likes}
                  </span>
                )}
                {comment.comment_replies_count > 0 && (
                  <span className="inline-flex items-center gap-1">
                    <MessageCircle className="h-3 w-3" />
                    {comment.comment_replies_count} replies
                  </span>
                )}
                {comment.reply_posted && (
                  <span className="inline-flex items-center gap-1 text-green-600 dark:text-green-400">
                    <CheckCircle2 className="h-3 w-3" />
                    Replied
                  </span>
                )}
              </div>
            </div>
          );
        })}
      </div>
    )}
  </CardContent>
</Card>
```
