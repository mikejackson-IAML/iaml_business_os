### Mutation Function

Create `dashboard/src/lib/api/linkedin-content-mutations.ts` with the `updateTopicStatus` function:

```typescript
import { getServerClient } from '@/lib/supabase/server';
import type { TopicRecommendationDb } from './linkedin-content-queries';

export async function updateTopicStatus(
  id: string,
  status: 'approved' | 'rejected' | 'pending'
): Promise<TopicRecommendationDb> {
  const supabase = getServerClient();

  const updateData: Record<string, unknown> = { status };

  if (status === 'approved') {
    updateData.approved_at = new Date().toISOString();
  } else {
    updateData.approved_at = null;
  }

  const { data, error } = await supabase
    .schema('linkedin_engine')
    .from('topic_recommendations')
    .update(updateData)
    .eq('id', id)
    .select()
    .single();

  if (error) {
    console.error('Topic status update error:', error);
    throw new Error('Failed to update topic status');
  }

  return data as TopicRecommendationDb;
}
```

NOTE on `.schema('linkedin_engine')`: The research identified that `.schema()` is the documented supabase-js API for custom schemas. The existing queries use dot notation (`.from('linkedin_engine.topic_recommendations')`) which was validated in Phase 1. For new code (mutations), use `.schema()`. If `.schema()` fails at runtime, fall back to dot notation: `.from('linkedin_engine.topic_recommendations')`.

### Add handleStatusChange Function

```typescript
async function handleStatusChange(topicId: string, newStatus: 'approved' | 'rejected') {
  const previousStatus = topicStatuses[topicId] || thisWeekTopics.find(t => t.id === topicId)?.status || 'pending';

  // Optimistic update
  setTopicStatuses(prev => ({ ...prev, [topicId]: newStatus }));

  try {
    const res = await fetch(`/api/linkedin-content/topics/${topicId}/status`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ status: newStatus }),
    });
    if (!res.ok) throw new Error('Failed to update');
    router.refresh(); // Revalidate server data
  } catch {
    // Revert on failure
    setTopicStatuses(prev => ({ ...prev, [topicId]: previousStatus }));
  }
}
```

### Add Helper Function for Effective Status

function getEffectiveStatus(topic: TopicRecommendationDb): string {
