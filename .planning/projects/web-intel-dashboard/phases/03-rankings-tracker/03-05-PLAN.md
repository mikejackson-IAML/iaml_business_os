```typescript
// Group rankings by keywordId for quick lookup
const rankingsByKeyword = useMemo(() => {
  const grouped = new Map<string, DailyRanking[]>();
  for (const r of rankings) {
    const existing = grouped.get(r.keywordId) || [];
    existing.push(r);
    grouped.set(r.keywordId, existing);
  }
  // Sort each group by date descending
  for (const [id, list] of grouped) {
    list.sort((a, b) => new Date(b.collectedDate).getTime() - new Date(a.collectedDate).getTime());
  }
  return grouped;
}, [rankings]);
```

3. For each keyword, compute:
   - currentPosition: First (most recent) ranking's position
   - positionChange: Difference between first and second ranking position
   - rankingHistory: Last 7 rankings as { date: string, position: number | null }[]

4. Replace inline row divs with:
```tsx
{sortedKeywords.map((kw) => {
  const kwRankings = rankingsByKeyword.get(kw.id) || [];
  const currentPosition = kwRankings[0]?.position ?? null;
  const previousPosition = kwRankings[1]?.position ?? null;
  const positionChange = (currentPosition !== null && previousPosition !== null)
    ? currentPosition - previousPosition
    : null;
  const rankingHistory = kwRankings.slice(0, 7).map(r => ({
    date: r.collectedDate.toISOString().split('T')[0],
    position: r.position,
  }));
  const latestRanking = kwRankings[0] ?? null;

  return (
    <KeywordRow
      key={kw.id}
      keyword={kw}
      currentPosition={currentPosition}
      positionChange={positionChange}
      rankingHistory={rankingHistory}
      latestRanking={latestRanking}
      isExpanded={expandedId === kw.id}
      onToggleExpand={() => setExpandedId(expandedId === kw.id ? null : kw.id)}
    />
  );
})}
```
