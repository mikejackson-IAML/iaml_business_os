import { createTask } from '@/lib/api/task-mutations';
import { AITaskSuggestion, AIAnalysisResult } from './ai-analysis-types';

interface CreateSuggestionResult {
  created: number;
  skipped: number;
  errors: string[];
}

/**
 * Create tasks from AI suggestions
 * Caps at maxSuggestions (default 10)
 * Uses dedupe_key to prevent duplicates
 */
export async function createAISuggestions(
  suggestions: AITaskSuggestion[],
  maxSuggestions: number = 10
): Promise<CreateSuggestionResult> {
  const weekNumber = getISOWeek(new Date());
  const year = new Date().getFullYear();
  let created = 0;
  let skipped = 0;
  const errors: string[] = [];

  // Cap at max suggestions
  const toCreate = suggestions.slice(0, maxSuggestions);

  for (const suggestion of toCreate) {
    const dedupeKey = generateSuggestionDedupeKey(suggestion, year, weekNumber);

    try {
      await createTask({
        title: suggestion.title,
        description: buildSuggestionDescription(suggestion),
        priority: suggestion.priority || 'normal',
        due_date: suggestion.due_date || null,
        source: 'ai',
        task_type: 'standard',
        dedupe_key: dedupeKey,
        ai_confidence: suggestion.confidence / 100,  // Convert 0-100 to 0-1
        ai_suggested_at: new Date().toISOString(),
      });
      created++;
    } catch (error) {
      if (error.message === 'DUPLICATE_DEDUPE_KEY') {
        skipped++;
      } else {
        errors.push(error.message);
      }
    }
  }

  return { created, skipped, errors };
}

function generateSuggestionDedupeKey(
  suggestion: AITaskSuggestion,
  year: number,
  week: number
): string {
  // Hash of title + type to detect duplicates within same week
  const hash = simpleHash(suggestion.title + suggestion.type);
  return `ai:${suggestion.type}:${year}-W${week}:${hash}`;
}

function buildSuggestionDescription(suggestion: AITaskSuggestion): string {
  let desc = suggestion.description || '';
  if (suggestion.confidence < 80 && suggestion.reasoning) {
    desc += `\n\n**Why AI suggests this:** ${suggestion.reasoning}`;
  }
  return desc;
}

function simpleHash(str: string): string {
  // Simple hash function for deduplication
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash).toString(16).slice(0, 8);
}
