<task id="3">
Add overdue severity levels to task-row.tsx:

```typescript
function getOverdueSeverity(task: TaskExtended): 'none' | 'mild' | 'severe' {
  if (!task.is_overdue || !task.due_date) return 'none';

  const daysOverdue = differenceInDays(new Date(), new Date(task.due_date));

  if (daysOverdue > 7) return 'severe';  // Red + pulse animation
  if (daysOverdue > 3) return 'mild';    // Darker red
  return 'none';  // Standard overdue styling
}
```
</task>
