```typescript
export async function setWeeklyGoalAction(target: number): Promise<void> {
  const user = await getCurrentUser();
  // Call action_center.set_user_weekly_goal RPC
  await supabase.rpc('set_user_weekly_goal', {
    p_user_id: user.id,
    p_target: target
  });
  revalidatePath('/dashboard/action-center');
  revalidatePath('/dashboard');
}
```
