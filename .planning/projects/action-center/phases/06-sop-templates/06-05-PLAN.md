## Objective

Create SOP detail page with tabs for Preview and Edit, following task detail page patterns.

## Context

The SOP detail page has two modes:
- **Preview tab**: View the SOP with mastery preview selector
- **Edit tab**: Manage SOP metadata and steps

Layout follows task detail pattern: two-column with content + sidebar.

import { Suspense } from "react";
import { notFound } from "next/navigation";
import { getSOPAction, getUserMasteryAction } from "../../sop-actions";
import { SOPDetailSkeleton } from "./sop-detail-skeleton";
import { SOPDetailContent } from "./sop-detail-content";

export async function generateMetadata({ params }: { params: { id: string } }) {
  const result = await getSOPAction(params.id);
  if (!result.success || !result.data) {
    return { title: "SOP Not Found" };
  }
  return { title: `${result.data.name} | SOP Templates` };
}

export default async function SOPDetailPage({
  params,
}: {
  params: { id: string };
}) {
  const [sopResult, masteryResult] = await Promise.all([
    getSOPAction(params.id),
    getUserMasteryAction(params.id),
  ]);

  if (!sopResult.success || !sopResult.data) {
    notFound();
  }

  return (
    <div className="container mx-auto p-6">
      <SOPDetailContent
        sop={sopResult.data}
        userMastery={masteryResult}
      />
    </div>
  );
}
