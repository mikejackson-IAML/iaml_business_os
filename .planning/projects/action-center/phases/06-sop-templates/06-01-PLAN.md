Add this function:
```typescript
export async function getUserMasteryForSOP(
  userId: string,
  sopId: string
): Promise<{ mastery_level: number; mastery_tier: string }> {
  const response = await fetch(
    `${API_BASE}/users/${userId}/mastery/${sopId}`,
    {
      headers: getAuthHeaders(),
      cache: "no-store",
    }
  );

  if (!response.ok) {
    // Default to novice if lookup fails
    return { mastery_level: 0, mastery_tier: 'novice' };
  }

  return response.json();
}
```

If the API endpoint doesn't exist, fall back to direct Supabase query:
```typescript
import { createClient } from "@/lib/supabase/server";

export async function getUserMasteryForSOP(
  userId: string,
  sopId: string
): Promise<{ mastery_level: number; mastery_tier: string }> {
  const supabase = await createClient();

  const { data, error } = await supabase
    .rpc('get_user_mastery', { p_user_id: userId, p_sop_id: sopId });

  if (error || data === null) {
    return { mastery_level: 0, mastery_tier: 'novice' };
  }

  const level = data as number;
  let tier = 'novice';
  if (level >= 10) tier = 'expert';
  else if (level >= 6) tier = 'proficient';
  else if (level >= 3) tier = 'developing';

  return { mastery_level: level, mastery_tier: tier };
}
```

<title>Add tasks-using-SOP query</title>
<instructions>
Add function to `dashboard/src/lib/api/sop-queries.ts` to get tasks referencing an SOP.

```typescript
export async function getTasksUsingSOP(sopId: string): Promise<{
  count: number;
  tasks: Array<{ id: string; title: string; status: string }>;
}> {
  const response = await fetch(
    `${API_BASE}/tasks?sop_template_id=${sopId}&limit=100`,
    {
      headers: getAuthHeaders(),
      cache: "no-store",
    }
  );

  if (!response.ok) {
    return { count: 0, tasks: [] };
  }

  const data = await response.json();
  return {
    count: data.meta?.total || data.data?.length || 0,
    tasks: data.data || [],
  };
}
```

This uses the existing tasks list endpoint with a filter. If the endpoint doesn't support `sop_template_id` filter, update the task list endpoint to support it.
