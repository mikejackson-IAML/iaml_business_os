<title>Update task detail page to fetch SOP and mastery</title>
<instructions>
Update `dashboard/src/app/dashboard/action-center/tasks/[id]/page.tsx` to fetch SOP data when task has an SOP reference.

```typescript
import { Suspense } from "react";
import { notFound } from "next/navigation";
import { getTaskAction } from "../../actions";
import { getSOPAction, getUserMasteryAction } from "../../sop-actions";
import { TaskDetailSkeleton } from "./task-detail-skeleton";
import { TaskDetailContent } from "./task-detail-content";

export default async function TaskDetailPage({
  params,
}: {
  params: { id: string };
}) {
  const taskResult = await getTaskAction(params.id);

  if (!taskResult.success || !taskResult.data) {
    notFound();
  }

  const task = taskResult.data;

  // Fetch SOP and mastery if task has SOP reference
  let sop = null;
  let sopMastery = { mastery_level: 0, mastery_tier: 'novice' as const };

  if (task.sop_template_id) {
    const [sopResult, masteryResult] = await Promise.all([
      getSOPAction(task.sop_template_id),
      getUserMasteryAction(task.sop_template_id),
    ]);

    if (sopResult.success && sopResult.data) {
      sop = sopResult.data;
    }
    if (masteryResult) {
      sopMastery = masteryResult;
    }
  }

  return (
    <div className="container mx-auto p-6">
      <TaskDetailContent
        task={task}
        sop={sop}
        sopMastery={sopMastery}
      />
    </div>
  );
}
```
</instructions>
</task>

```typescript
// In task-detail-content.tsx or a separate utils file

function buildVariableContext(task: TaskExtended): Record<string, string> {
  const context: Record<string, string> = {};

  // Task-level variables
  context.task_title = task.title;
  context.task_id = task.id;

  // Related entity variables (if exists)
  if (task.related_entity_type && task.related_entity_id) {
    context.entity_type = task.related_entity_type;
    context.entity_id = task.related_entity_id;

    // If we have denormalized entity data, add it
    // This would come from the task API joining related entity
    if (task.related_entity_name) {
      context.entity_name = task.related_entity_name;
    }
  }

  // Common entity type mappings
  if (task.related_entity_type === 'program_instance') {
    context.program_name = task.related_entity_name || 'Unknown Program';
    // Add more program-specific fields if available
  }

  if (task.related_entity_type === 'campaign') {
    context.campaign_name = task.related_entity_name || 'Unknown Campaign';
  }

  // Department context
  if (task.department) {
    context.department = task.department;
  }

  // Assignee context
  if (task.assignee_name) {
    context.assignee_name = task.assignee_name;
  }

  return context;
}
```

The variable context is built from:
1. Task's own data (title, id, department)
2. Related entity data (if task is linked to a program, campaign, etc.)
3. Assignee information

Variables not found in context will remain as `{{variable_name}}` in the output.
