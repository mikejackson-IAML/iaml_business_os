# Plan 08-02: Alert Accumulation Tracking

## Objective

Create a table to track alert occurrences for the "3x in 24 hours" info alert rule. This enables detecting when low-priority alerts accumulate enough to warrant task creation.

## Context

From 08-CONTEXT.md:
- Exception 2: Accumulation rule - same info alert 3+ times in 24 hours creates a low priority task
- Accumulation detection needs a sliding window: "3 times in last 24 hours" not "3 times today"

```sql
-- Alert Integration: Alert Accumulation Tracking
-- Migration: Track alert occurrences for 3x/24h rule
-- Date: 2026-01-24

-- ============================================
-- ALERT OCCURRENCES TABLE
-- Tracks individual alert occurrences for accumulation detection
-- ============================================
CREATE TABLE IF NOT EXISTS action_center.alert_occurrences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  -- Alert identification (composite key for grouping)
  alert_type TEXT NOT NULL,
  affected_resource TEXT NOT NULL,  -- e.g., 'www.iaml.com', 'instructor:uuid'

  -- Alert details (for context)
  severity TEXT NOT NULL CHECK (severity IN ('critical', 'warning', 'info')),
  original_title TEXT NOT NULL,

  -- Source tracking
  source_alert_id TEXT,  -- ID from originating system
  source_system TEXT,    -- e.g., 'faculty_scheduler', 'ssl_monitor'

  -- Timestamp
  occurred_at TIMESTAMPTZ DEFAULT NOW(),

  -- Whether this occurrence resulted in a task
  task_created BOOLEAN DEFAULT FALSE,
  task_id UUID  -- References action_center.tasks(id)
);

-- Indexes for efficient accumulation queries
CREATE INDEX IF NOT EXISTS idx_alert_occurrences_lookup
  ON action_center.alert_occurrences(alert_type, affected_resource, occurred_at DESC);
CREATE INDEX IF NOT EXISTS idx_alert_occurrences_recent
  ON action_center.alert_occurrences(occurred_at DESC);

-- ============================================
-- FUNCTION: Check Alert Accumulation
-- Returns true if accumulation threshold reached
-- ============================================
CREATE OR REPLACE FUNCTION action_center.check_alert_accumulation(
  p_alert_type TEXT,
  p_affected_resource TEXT,
  p_window_hours INTEGER DEFAULT 24,
  p_threshold INTEGER DEFAULT 3
)
RETURNS TABLE (
  should_create_task BOOLEAN,
  occurrence_count INTEGER,
  oldest_in_window TIMESTAMPTZ
) AS $$
DECLARE
  v_count INTEGER;
  v_oldest TIMESTAMPTZ;
  v_window_start TIMESTAMPTZ;
BEGIN
  v_window_start := NOW() - (p_window_hours || ' hours')::INTERVAL;

  SELECT COUNT(*), MIN(occurred_at)
  INTO v_count, v_oldest
  FROM action_center.alert_occurrences
  WHERE alert_type = p_alert_type
    AND affected_resource = p_affected_resource
    AND occurred_at >= v_window_start
    AND task_created = FALSE;  -- Only count those that didn't already trigger a task

  RETURN QUERY SELECT
    (v_count >= p_threshold) AS should_create_task,
    v_count AS occurrence_count,
    v_oldest AS oldest_in_window;
END;
$$ LANGUAGE plpgsql STABLE;

-- ============================================
-- FUNCTION: Record Alert Occurrence
-- Logs an alert occurrence and returns accumulation status
-- ============================================
CREATE OR REPLACE FUNCTION action_center.record_alert_occurrence(
  p_alert_type TEXT,
  p_affected_resource TEXT,
  p_severity TEXT,
  p_original_title TEXT,
  p_source_alert_id TEXT DEFAULT NULL,
  p_source_system TEXT DEFAULT NULL
)
RETURNS TABLE (
  occurrence_id UUID,
  should_create_task BOOLEAN,
  occurrence_count INTEGER
) AS $$
DECLARE
  v_occurrence_id UUID;
  v_config action_center.alert_config%ROWTYPE;
  v_accumulation RECORD;
BEGIN
  -- Get config for this alert type
  SELECT * INTO v_config
  FROM action_center.alert_config
  WHERE alert_type = p_alert_type;

  -- Insert occurrence
  INSERT INTO action_center.alert_occurrences (
    alert_type,
    affected_resource,
    severity,
    original_title,
    source_alert_id,
    source_system
  ) VALUES (
    p_alert_type,
    p_affected_resource,
    p_severity,
    p_original_title,
    p_source_alert_id,
    p_source_system
  )
  RETURNING id INTO v_occurrence_id;

  -- Check accumulation if this is an info alert
  IF p_severity = 'info' AND v_config.id IS NOT NULL THEN
    SELECT * INTO v_accumulation
    FROM action_center.check_alert_accumulation(
      p_alert_type,
      p_affected_resource,
      COALESCE(v_config.accumulation_window_hours, 24),
      COALESCE(v_config.accumulation_threshold, 3)
    );

    RETURN QUERY SELECT
      v_occurrence_id,
      v_accumulation.should_create_task,
      v_accumulation.occurrence_count;
  ELSE
    RETURN QUERY SELECT
      v_occurrence_id,
      FALSE,
      1;
  END IF;
END;
$$ LANGUAGE plpgsql;

-- ============================================
-- FUNCTION: Mark Occurrences as Task Created
-- Called after creating a task from accumulated alerts
-- ============================================
CREATE OR REPLACE FUNCTION action_center.mark_occurrences_task_created(
  p_alert_type TEXT,
  p_affected_resource TEXT,
  p_task_id UUID,
  p_window_hours INTEGER DEFAULT 24
)
RETURNS INTEGER AS $$
DECLARE
  v_count INTEGER;
BEGIN
  UPDATE action_center.alert_occurrences
  SET
    task_created = TRUE,
    task_id = p_task_id
  WHERE alert_type = p_alert_type
    AND affected_resource = p_affected_resource
    AND occurred_at >= NOW() - (p_window_hours || ' hours')::INTERVAL
    AND task_created = FALSE;

  GET DIAGNOSTICS v_count = ROW_COUNT;
  RETURN v_count;
END;
$$ LANGUAGE plpgsql;

-- ============================================
-- CLEANUP: Purge old occurrences (run weekly)
-- Keeps 30 days of history
-- ============================================
CREATE OR REPLACE FUNCTION action_center.purge_old_alert_occurrences()
RETURNS INTEGER AS $$
DECLARE
  v_count INTEGER;
BEGIN
  DELETE FROM action_center.alert_occurrences
  WHERE occurred_at < NOW() - INTERVAL '30 days';

  GET DIAGNOSTICS v_count = ROW_COUNT;
  RETURN v_count;
END;
$$ LANGUAGE plpgsql;

-- ============================================
-- COMMENTS
-- ============================================
COMMENT ON TABLE action_center.alert_occurrences IS
  'Tracks individual alert occurrences for accumulation detection (3x/24h rule)';
COMMENT ON FUNCTION action_center.check_alert_accumulation IS
  'Returns whether accumulation threshold has been reached for an alert type + resource';
COMMENT ON FUNCTION action_center.record_alert_occurrence IS
  'Logs alert occurrence and returns whether to create a task based on accumulation';
```
